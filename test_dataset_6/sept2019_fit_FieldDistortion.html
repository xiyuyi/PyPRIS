Find the peak coordiantes, and mark them in the 0th order diffraction image.

# detect 3D peaks
im_list = list()
peaks_list = list()
title_list = ['calib1 (w CL)','calib2 (w CL)','calib3 (wo CL)']


im_list.append(np.asarray(calib1[2][:,:]))
im_list.append(np.asarray(calib2[2][:,:]))
im_list.append(np.asarray(calib3_woCL[2][:,:]))

# image_max is the dilation of im with a 2*2 structuring element
# It is used within peak_local_max function
for im in im_list:
    image_max = ndi.maximum_filter(im.T, size=10, mode='constant') # the mode option doesn't seem to play a significant role.

    # Comparison between image_max and im to find the coordinates of local maxima
    coordinates = peak_local_max(im.T, min_distance=8) # find peak after transpose, so the peak orders won't be messed up.
    peaks = np.column_stack((coordinates, im.T[coordinates[:,0],coordinates[:,1]]))
    peaks = peaks[np.where(peaks[:,2]>10000)] # get only the desired peaks with given intensity. peaks[:,2] is the intensities
    peaks = peaks[np.where(peaks[:,1]>280)] # get only the bottom section
    peaks = peaks[:,[1, 0]] # now swap columes because the peak was detected with the transposed image.
    peaks_list.append(copy.deepcopy(peaks))


# display results
fig, axes = plt.subplots(1, 3, figsize=(18, 18), sharex=True, sharey=True)
i = 0
ax = axes.ravel()
for im, title, peaks in zip(im_list, title_list, peaks_list):
    ax[i].imshow(im, cmap=plt.cm.gray)
    ax[i].autoscale(False)
    ax[i].plot(peaks[:, 1], peaks[:, 0], 'r.') # 
    ax[i].axis('off')
    ax[i].set_title(title, fontsize=26)
    i += 1
    
fig.tight_layout()
plt.show()

def xy_woCL(xy_wCL, A, B): # 'straight line' y=f(x)
    return A*xy_wCL + B
